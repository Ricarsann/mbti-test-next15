**方法 2：使用 Hook 自定义**

```tsx
'use client';

import { useTest } from '@/src/hooks/useTest';

export default function MyTest() {
  const {
    currentQuestion,
    currentResponse,
    progress,
    answerQuestion,
    nextQuestion,
  } = useTest();

  return (
    <div>
      <p>进度: {progress}%</p>
      <h2>{currentQuestion?.text}</h2>
      {[1, 2, 3, 4, 5].map((value) => (
        <button
          key={value}
          onClick={() => {
            if (currentQuestion) {
              answerQuestion(currentQuestion.id, value);
              nextQuestion();
            }
          }}
        >
          {value}
        </button>
      ))}
    </div>
  );
}
```
